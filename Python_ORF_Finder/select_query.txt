#select all van tabbel RESULT
SELECT *
FROM RESULT;

#select columns vanuit ORF als de fk frame 1 is
SELECT ID, NUCLEOTIDE_SEQUENTIE, POSITION, FRAME_FRAME
FROM ORF
WHERE FRAME_FRAME=1;

#select alle ORF's met organisme ID 1
SELECT ORF.ID, ORF.POSITION, ORF.FRAME_FRAME, ORGANISME.ID, ORGANISME.HEADER
FROM ORF
INNER JOIN ORGANISME ON ORF.ORGANISME_ID=ORGANISME.ID
WHERE ORF.ORGANISME_ID=1;

#select ORF's waar interest_gene id 1 is gesorteerd op blast resultaat score
SELECT RESULT.ID, RESULT.EXPECT, RESULT.SCORE, RESULT.ORF_ID, RESULT.INTEREST_GENES_ID, INTEREST_GENES.HEADER, ORF.POSITIE
FROM RESULT
INNER JOIN INTEREST_GENES ON RESULT.INTEREST_GENES_ID=INTEREST_GENES.ID
INNER JOIN ORF ON RESULT.ORF_ID=ORF.ID
WHERE INTEREST_GENES.ID=1
ORDER BY RESULT.SCORE DESC;
